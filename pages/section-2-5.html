Read this if pointers, memory, or structs feel uncertain. Skip to Section 3 if all five concepts are solid.
2.5.1 Memory: Stack vs Heap
void demonstrate_memory() {
    // STACK — automatic, fast, limited (~1–8MB)
    int altitude = 100;
    float battery[4] = {87.2, 86.1, 85.0, 83.9};
    // freed automatically when function returns

    // HEAP — manual, slower, much larger
    float* sensor_readings = malloc(10 * sizeof(float));
    sensor_readings[0] = 23.8;
    free(sensor_readings);   // MUST free — forget = memory leak
    sensor_readings = NULL;
}
Never return a pointer to a stack variable. 
float* bad()  { float x = 100.0; return &x; }  // WRONG — dangling pointer
float* good() { float* x = malloc(sizeof(float)); *x = 100.0; return x; }

Why this matters: C++ smart pointers (Section 3.4) automate free() — but only make sense if the underlying concept is understood.

2.5.2 Pointers
float altitude = 100.0;
float* ptr = &altitude;       // & = "address of"
printf("%.1f\n", *ptr);       // 100.0  (* = dereference)
*ptr = 200.0;
printf("%.1f\n", altitude);   // 200.0 — original changed

// NULL — always check before dereferencing
float* empty = NULL;
if (empty != NULL) { printf("%.1f\n", *empty); }
// Dereferencing NULL = segfault = most common crash
// Pointer arithmetic — used in sensor data buffers
float readings[5] = {10.0, 20.0, 30.0, 40.0, 50.0};
float* p = readings;
printf("%.1f\n", *(p+1));    // 20.0
2.5.3 Structs
typedef struct {
    double latitude;
    double longitude;
    float  altitude;
} GPSPosition;

void update_altitude(GPSPosition* pos, float new_alt) {
    pos->altitude = new_alt;  // -> for pointer to struct
}

GPSPosition home = {23.8103, 90.4125, 10.0};
update_altitude(&home, 25.0);
printf("%.1f\n", home.altitude);  // 25.0

Why this matters: Every ROS2 message works exactly like this struct. sensor_msgs::msg::NavSatFix has latitude, longitude, altitude fields — identical concept.

2.5.4 Functions and Scope
// Pass by value — original unchanged
void reset_by_value(float altitude) { altitude = 0.0; }

// Pass by pointer — modifies original
void reset_by_pointer(float* altitude) { *altitude = 0.0; }

float alt = 100.0;
reset_by_value(alt);    printf("%.1f\n", alt);  // 100.0
reset_by_pointer(&alt); printf("%.1f\n", alt);  // 0.0
// Output parameter — returning multiple values
void get_gps_error(float* h_err, float* v_err) {
    *h_err = 2.5;  *v_err = 4.1;
}
float h, v;
get_gps_error(&h, &v);  // h=2.5, v=4.1
2.5.5 Header Files and Compilation Units
// gps.h — declares what exists
#ifndef GPS_H
#define GPS_H
typedef struct { double lat; double lon; float alt; } GPSPos;
float distance(GPSPos a, GPSPos b);
#endif
// gps.c — defines how it works
#include "gps.h"
#include <math.h>
float distance(GPSPos a, GPSPos b) {
    double dlat = a.lat - b.lat, dlon = a.lon - b.lon;
    return (float)sqrt(dlat*dlat + dlon*dlon);
}
gcc main.c gps.c -o drone_nav -lm

Why this matters: CMakeLists.txt does exactly this — specifies which .cpp files to compile and which libraries to link.
