Every ROS2 C++ node follows this exact structure. Learn it once — it applies everywhere.

5.1 Universal Node Pattern
#include "rclcpp/rclcpp.hpp"
#include "some_msgs/msg/some_type.hpp"

// Step 1: Inherit from rclcpp::Node
class MyNode : public rclcpp::Node {
public:
    // Step 2: Constructor — set up publishers, subscribers, timers
    MyNode() : Node("my_node_name") {
        publisher_ = create_publisher<MsgType>("/topic", 10);

        subscriber_ = create_subscription<MsgType>(
            "/other_topic", 10,
            [this](const MsgType::SharedPtr msg) { on_message(msg); }
        );

        timer_ = create_wall_timer(
            std::chrono::milliseconds(100),
            [this]() { on_timer(); }
        );
        RCLCPP_INFO(get_logger(), "Node started");
    }

private:
    // Step 3: Callbacks
    void on_message(const MsgType::SharedPtr msg) { /* handle */ }
    void on_timer() { publisher_->publish(MsgType()); }

    // Step 4: Members — always SharedPtr
    rclcpp::Publisher<MsgType>::SharedPtr publisher_;
    rclcpp::Subscription<MsgType>::SharedPtr subscriber_;
    rclcpp::TimerBase::SharedPtr timer_;
};

// Step 5: main — always identical
int main(int argc, char* argv[]) {
    rclcpp::init(argc, argv);
    rclcpp::spin(std::make_shared<MyNode>());
    rclcpp::shutdown();
    return 0;
}
The 5-step pattern to memorize:
1. Inherit from rclcpp::Node
2. Constructor sets up pub/sub/timers
3. Private callbacks handle data
4. Members are always SharedPtr
5. main() is always the same 3 lines
5.2 CMakeLists.txt + package.xml Templates
Most common mistake: adding a new message type to the .cpp file but forgetting to add it to BOTH CMakeLists.txt AND package.xml. Always update both together.
cmake_minimum_required(VERSION 3.8)
project(my_package)

find_package(ament_cmake REQUIRED)
find_package(rclcpp REQUIRED)
find_package(sensor_msgs REQUIRED)   # ← add every msg package used

add_executable(my_node src/my_node.cpp)
ament_target_dependencies(my_node rclcpp sensor_msgs)
install(TARGETS my_node DESTINATION lib/${PROJECT_NAME})
ament_package()
<!-- package.xml -->
<depend>rclcpp</depend>
<depend>sensor_msgs</depend>
5.3 Build and Run Workflow
cd ~/drone_ws
colcon build --packages-select my_package
source install/setup.bash          # don't forget this
ros2 run my_package my_node

# Verify in another terminal:
ros2 topic echo /topic_name
ros2 node list
Common mistake: forgetting source install/setup.bash after build. ROS2 won't find your node until you source.
