Project 1 — GPS Telemetry Publisher  1 week

What it does: Publishes simulated GPS data at 10Hz on /gps/fix.
Concepts: Node structure, publishers, timers, sensor message types.

ros2 pkg create --build-type ament_cmake telemetry_node \
  --dependencies rclcpp sensor_msgs
class TelemetryNode : public rclcpp::Node {
public:
    TelemetryNode() : Node("telemetry") {
        publisher_ = create_publisher<sensor_msgs::msg::NavSatFix>("/gps/fix", 10);
        timer_ = create_wall_timer(100ms, [this]() { publish_gps(); });
    }
private:
    void publish_gps() {
        auto msg = sensor_msgs::msg::NavSatFix();
        msg.latitude  = 23.8103 + (counter_ * 0.0001);
        msg.longitude = 90.4125 + (counter_ * 0.0001);
        msg.altitude  = 100.0;
        publisher_->publish(msg);
        counter_++;
    }
    rclcpp::Publisher<sensor_msgs::msg::NavSatFix>::SharedPtr publisher_;
    rclcpp::TimerBase::SharedPtr timer_;
    int counter_ = 0;
};
Project 2 — Obstacle Detection Subscriber  1 week

What it does: Reads lidar data, logs warning when obstacle is within 5m.
Concepts: Subscribers, callbacks, sensor array processing, invalid reading filtering.

void process_scan(const sensor_msgs::msg::LaserScan::SharedPtr msg) {
    float closest = std::numeric_limits<float>::max();
    for (float r : msg->ranges) {
        if (std::isfinite(r) && r >= msg->range_min && r <= msg->range_max)
            closest = std::min(closest, r);
    }
    if (closest < 5.0f)
        RCLCPP_WARN(get_logger(), "OBSTACLE at %.2fm!", closest);
}

Project 4 — PID Altitude Controller  1 week

What it does: Maintains target altitude using a PID controller.
Concepts: Control theory, real-time constraints, anti-windup.

class PIDController {
public:
    PIDController(float kp, float ki, float kd, float min, float max)
        : kp_(kp), ki_(ki), kd_(kd), min_(min), max_(max) {}

    float compute(float setpoint, float measurement, float dt) {
        float error = setpoint - measurement;
        integral_ = std::clamp(integral_ + error * dt, -10.0f, 10.0f);
        float d = (dt > 0) ? (error - prev_error_) / dt : 0;
        prev_error_ = error;
        return std::clamp(kp_*error + ki_*integral_ + kd_*d, min_, max_);
    }
private:
    float kp_, ki_, kd_, min_, max_;
    float integral_ = 0, prev_error_ = 0;
};

PID Tuning Guide

1. Start: Kp=1.0, Ki=0.0, Kd=0.0
2. Increase Kp until fast but oscillating
3. Add Kd=0.1 to dampen oscillation
4. Add Ki=0.001 only if steady-state error remains
